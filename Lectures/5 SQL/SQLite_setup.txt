npm install touch-cli -g

#creates file
touch flights.sql

sqlite3 flights.sql

CREATE TABLE flights(
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    origin TEXT NOT NULL,
    destination TEXT NOT NULL,
    duration INTEGER NOT NULL
);

#show tables
.tables
